## 一、V1 版本概述

### 1.1 版本目标

构建系统的最小可行性版本（MVP），实现基础的反向代理功能与URL重写核心，为后续版本迭代奠定基础。

### 1.2 核心价值

- **完整的反向代理能力**：支持HTTP/HTTPS协议的请求转发
- **精准的URL修正引擎**：确保所有资源请求都通过代理服务器
- **基础性能保障**：单节点支持2000+并发连接，URL处理延迟<50ms
- **可观测性**：完善的日志系统与监控面板

### 2.1 程序主入口 (SilkRoad.py)

#### 功能说明

作为整个系统的启动入口，负责：

- 初始化所有核心模块
- 启动代理服务器
- 注册信号处理器
- 协调各模块生命周期

#### 技术实现

```python
import asyncio
import signal
from modules.cfg import ConfigManager
from modules.proxy import ProxyServer
from modules.logging import Logger
from modules.command import CommandHandler
from modules.exit import GracefulExit

class SilkRoad:
    def __init__(self):
        self.config = ConfigManager()
        self.logger = Logger(self.config)
        self.proxy_server = None
        self.command_handler = None
        self.shutdown_event = asyncio.Event()
        
    async def initialize(self):
        """初始化所有模块"""
        await self.config.load()
        self.logger.info("配置加载完成")
        
        self.proxy_server = ProxyServer(
            host=self.config.get('host', '0.0.0.0'),
            port=self.config.get('port', 8080),
            config=self.config,
            logger=self.logger
        )
        
        self.command_handler = CommandHandler(
            proxy_server=self.proxy_server,
            config=self.config,
            logger=self.logger
        )
        
        GracefulExit.setup(self.shutdown_event, self.logger)
        
    async def run(self):
        """启动主服务"""
        await self.initialize()
        
        tasks = [
            asyncio.create_task(self.proxy_server.start()),
            asyncio.create_task(self.command_handler.start()),
            asyncio.create_task(self.wait_for_shutdown())
        ]
        
        await asyncio.gather(*tasks)
        
    async def wait_for_shutdown(self):
        """等待关闭信号"""
        await self.shutdown_event.wait()
        self.logger.info("开始优雅关闭...")
        await self.proxy_server.stop()
        await self.command_handler.stop()
        await self.logger.close()

if __name__ == "__main__":
    app = SilkRoad()
    asyncio.run(app.run())
```

```python
import json
import os
from pathlib import Path
from typing import Any, Dict
from watchdog.observers import Observer
from watchdog.events import FileSystemEventHandler

class ConfigManager:
    def __init__(self, config_path: str = "databases/config.json"):
        self.config_path = config_path
        self.config: Dict[str, Any] = {}
        self.default_config = self._get_default_config()
        self._callbacks = []
        
    async def load(self) -> None:
        """加载配置文件"""
        if not os.path.exists(self.config_path):
            self.config = self.default_config
            await self._save_default_config()
        else:
            try:
                with open(self.config_path, 'r', encoding='utf-8') as f:
                    self.config = json.load(f)
                self._validate_config()
            except json.JSONDecodeError as e:
                raise ConfigError(f"配置文件格式错误: {e}")
                
    def get(self, key: str, default: Any = None) -> Any:
        """获取配置项（支持点分隔符）"""
        keys = key.split('.')
        value = self.config
        for k in keys:
            if isinstance(value, dict):
                value = value.get(k)
            else:
                return default
        return value if value is not None else default
        
    def _get_default_config(self) -> Dict[str, Any]:
        """获取默认配置"""
        return {
            "server": {
                "host": "0.0.0.0",
                "port": 8080,
                "backlog": 2048,
                "maxConnections": 2000
            },
            "proxy": {
                "timeout": 30,
                "maxRetries": 3,
                "enableCompression": True,
                "compressionLevel": 6
            },
            "logging": {
                "level": "INFO",
                "rotation": "1 day",
                "retention": "30 days"
            },
            "pageRoutes": {
                "/": "main",
                "/admin": "admin",
                "/error": "error"
            }
        }
        
    def _validate_config(self) -> None:
        """验证配置项"""
        required_keys = ['server.host', 'server.port']
        for key in required_keys:
            if self.get(key) is None:
                raise ConfigError(f"缺少必需配置项: {key}")
                
        # 验证端口范围
        port = self.get('server.port')
        if not (1 <= port <= 65535):
            raise ConfigError(f"端口号无效: {port}")
            
    def register_callback(self, callback):
        """注册配置变更回调（为热重载准备）"""
        self._callbacks.append(callback)
```

### 2.3 核心代理转发引擎 (modules/proxy.py)

#### 功能说明

实现反向代理的核心逻辑，包括：

- 接收客户端请求
- 转发请求到目标服务器
- 处理响应并返回给客户端
- 协调URL修正模块

#### 技术实现

```python
import asyncio
import aiohttp
from typing import Optional
from urllib.parse import urlparse
from modules.url.handle import URLHandler
from modules.ua import UAHandler

class ProxyServer:
    def __init__(self, host: str, port: int, config, logger):
        self.host = host
        self.port = port
        self.config = config
        self.logger = logger
        self.server = None
        self.session = None
        self.url_handler = URLHandler(config, logger)
        self.ua_handler = UAHandler()
        self.active_connections = 0
        self.is_running = False
        
    async def start(self):
        """启动代理服务器"""
        self.session = aiohttp.ClientSession(
            timeout=aiohttp.ClientTimeout(total=self.config.get('proxy.timeout', 30)),
            connector=aiohttp.TCPConnector(limit=100)
        )
        
        self.server = await asyncio.start_server(
            self._handle_connection,
            self.host,
            self.port,
            backlog=self.config.get('server.backlog', 2048)
        )
        
        self.is_running = True
        self.logger.info(f"代理服务器启动: {self.host}:{self.port}")
        
        async with self.server:
            await self.server.serve_forever()
            
    async def stop(self):
        """停止服务器"""
        self.is_running = False
        if self.server:
            self.server.close()
            await self.server.wait_closed()
        if self.session:
            await self.session.close()
            
    async def _handle_connection(self, reader, writer):
        """处理客户端连接"""
        self.active_connections += 1
        try:
            await self._process_request(reader, writer)
        except Exception as e:
            self.logger.error(f"处理请求错误: {e}")
        finally:
            self.active_connections -= 1
            writer.close()
            await writer.wait_closed()
            
    async def _process_request(self, reader, writer):
        """处理HTTP请求"""
        # 1. 解析请求行
        request_line = await reader.readline()
        if not request_line:
            return
            
        method, path, version = self._parse_request_line(request_line)
        
        # 2. 解析请求头
        headers = await self._parse_headers(reader)
        
        # 3. 解析目标URL
        target_url = self._parse_target_url(path, headers)
        if not target_url:
            await self._send_error(writer, 400, "Invalid URL")
            return
            
        # 4. 读取请求体（如果有）
        body = None
        if method in ['POST', 'PUT', 'PATCH']:
            content_length = int(headers.get('Content-Length', 0))
            if content_length > 0:
                body = await reader.read(content_length)
                
        # 5. 构建转发请求
        forward_headers = self._build_forward_headers(headers, target_url)
        
        # 6. 发送请求到目标服务器
        try:
            async with self.session.request(
                method,
                target_url,
                headers=forward_headers,
                data=body
            ) as response:
                await self._send_response(writer, response, target_url)
        except asyncio.TimeoutError:
            await self._send_error(writer, 504, "Gateway Timeout")
        except aiohttp.ClientError as e:
            self.logger.error(f"目标服务器错误: {e}")
            await self._send_error(writer, 502, "Bad Gateway")
            
    def _parse_request_line(self, request_line: bytes) -> tuple:
        """解析请求行"""
        parts = request_line.decode('utf-8').strip().split(' ')
        if len(parts) != 3:
            raise ValueError("Invalid request line")
        return parts[0], parts[1], parts[2]
        
    def _parse_target_url(self, path: str, headers: dict) -> Optional[str]:
        """解析目标URL"""
        # 处理绝对路径（标准代理格式）
        if path.startswith('http://') or path.startswith('https://'):
            return path
            
        # 处理相对路径（反向代理格式）
        # 从Host头或路径中提取目标域名
        host = headers.get('Host', '')
        if '/' in path:
            # 路径格式: /target.com/path
            parts = path.lstrip('/').split('/', 1)
            target_host = parts[0]
            target_path = '/' + parts[1] if len(parts) > 1 else '/'
        else:
            target_host = host
            target_path = path
            
        # 补全协议
        if not target_host.startswith('http'):
            target_host = 'https://' + target_host
            
        return f"{target_host}{target_path}"
        
    def _build_forward_headers(self, original_headers: dict, target_url: str) -> dict:
        """构建转发请求头"""
        forward_headers = {}
        
        # 复制原始请求头
        skip_headers = {'Host', 'Connection', 'Accept-Encoding'}
        for key, value in original_headers.items():
            if key not in skip_headers:
                forward_headers[key] = value
                
        # 设置新的请求头
        parsed = urlparse(target_url)
        forward_headers['Host'] = parsed.netloc
        forward_headers['Connection'] = 'keep-alive'
        
        # 仅支持gzip和deflate
        forward_headers['Accept-Encoding'] = 'gzip, deflate'
        
        # UA随机化
        forward_headers['User-Agent'] = self.ua_handler.get_random_ua()
        
        return forward_headers
        
    async def _send_response(self, writer, response, target_url: str):
        """发送响应给客户端"""
        # 1. 读取响应体
        content = await response.read()
        
        # 2. 解压缩
        content = await self._decompress_content(
            content,
            response.headers.get('Content-Encoding')
        )
        
        # 3. URL修正
        content_type = response.headers.get('Content-Type', '')
        if self._should_rewrite(content_type):
            content = await self.url_handler.rewrite(
                content,
                content_type,
                target_url
            )
            
        # 4. 重新压缩
        content, encoding = await self._compress_content(content)
        
        # 5. 构建响应头
        headers = dict(response.headers)
        headers['Content-Length'] = str(len(content))
        headers['Content-Encoding'] = encoding
        headers['Via'] = 'SilkRoad-Next/1.0'
        
        # 6. 发送响应
        status_line = f"HTTP/1.1 {response.status} {response.reason}\r\n"
        writer.write(status_line.encode('utf-8'))
        
        for key, value in headers.items():
            writer.write(f"{key}: {value}\r\n".encode('utf-8'))
        writer.write(b"\r\n")
        writer.write(content)
        await writer.drain()
```

async def _process_request(self, reader, writer):
    max_redirects = self.config.get('urlRewrite.maxRedirects', 10)
    redirect_count = 0
    current_url = target_url
    
    while redirect_count < max_redirects:

            break
    else:

### 2.4 URL修正核心模块 (modules/url/)

#### 2.4.1 URL处理入口 (handle.py)

```python
import re
from typing import Tuple, Optional
from urllib.parse import urlparse, urljoin
import gzip
import zlib
import chardet

class URLHandler:
    def __init__(self, config, logger):
        self.config = config
        self.logger = logger
        self.handlers = {
            'text/html': HTMLHandler(),
            'text/css': CSSHandler(),
            'application/javascript': JSHandler(),
            'application/xml': XMLHandler(),
            'application/json': JSONHandler(),
        }
        
        # 预编译正则表达式
        self._compile_patterns()
        
    def _compile_patterns(self):
        """预编译所有正则表达式"""
        # URL模式
        self.absolute_url_pattern = re.compile(
            r'https?://[^\s<>"\'\)]+',
            re.IGNORECASE
        )
        
        # 相对URL模式
        self.relative_url_pattern = re.compile(
            r'(href|src|action)=["\']([^"\']+)["\']',
            re.IGNORECASE
        )
        
    async def rewrite(self, content: bytes, content_type: str, 
                     base_url: str) -> bytes:
        """重写URL"""
        # 1. 检测编码
        encoding = self._detect_encoding(content, content_type)
        
        # 2. 解码为字符串
        try:
            text = content.decode(encoding)
        except UnicodeDecodeError:
            # 如果解码失败，尝试其他编码
            text = content.decode('utf-8', errors='ignore')
            
        # 3. 获取对应的处理器
        handler = self._get_handler(content_type)
        if not handler:
            return content
            
        # 4. 执行URL重写
        rewritten_text = await handler.rewrite(text, base_url, self.config)
        
        # 5. 重新编码
        return rewritten_text.encode(encoding)
        
    def _detect_encoding(self, content: bytes, content_type: str) -> str:
        """检测内容编码"""
        # 从Content-Type中提取
        if 'charset=' in content_type:
            charset = content_type.split('charset=')[-1].strip()
            return charset
            
        # 使用chardet检测
        detected = chardet.detect(content)
        return detected.get('encoding', 'utf-8')
        
    def _get_handler(self, content_type: str):
        """获取内容类型对应的处理器"""
        for ct, handler in self.handlers.items():
            if ct in content_type:
                return handler
        return None
```

            re.compile(r'<source\s+[^>]*src=["\']([^"\']+)["\']', re.IGNORECASE),
            # <embed src="...">
            re.compile(r'<embed\s+[^>]*src=["\']([^"\']+)["\']', re.IGNORECASE),
        ]
        
        # 内联样式中的URL
        self.style_url_pattern = re.compile(
            r'url\(["\']?([^)"\'\s]+)["\']?\)',
            re.IGNORECASE
        )
        
        # srcset属性（响应式图片）
        self.srcset_pattern = re.compile(
            r'srcset=["\']([^"\']+)["\']',
            re.IGNORECASE
        )
        
    async def rewrite(self, html: str, base_url: str, config) -> str:
        """重写HTML中的URL"""
        # 1. 处理标签属性
        for pattern in self.tag_patterns:
            html = await self._rewrite_tag_urls(html, pattern, base_url)
            
        # 2. 处理内联样式
        html = await self._rewrite_style_urls(html, base_url)
        
        # 3. 处理srcset
        html = await self._rewrite_srcset(html, base_url)
        
        # 4. 处理base标签
        html = await self._handle_base_tag(html, base_url)
        
        return html
        
    async def _rewrite_tag_urls(self, html: str, pattern, base_url: str) -> str:
        """重写标签中的URL"""
        def replace_url(match):
            original_url = match.group(1)
            rewritten_url = self._rewrite_single_url(original_url, base_url)
            return match.group(0).replace(original_url, rewritten_url)
            
        return pattern.sub(replace_url, html)
        
    def _rewrite_single_url(self, url: str, base_url: str) -> str:
        """重写单个URL"""
        # 1. 跳过特殊协议
        if url.startswith(('javascript:', 'mailto:', 'tel:', 'data:', '#')):
            return url
            
        # 2. 补全相对URL
        if not url.startswith(('http://', 'https://')):
            url = urljoin(base_url, url)
            
        # 3. 转换为代理URL
        return self._to_proxy_url(url)
        
    def _to_proxy_url(self, target_url: str) -> str:
        """将目标URL转换为代理URL"""
        parsed = urlparse(target_url)
        
        # 格式: /target_host/path?query#fragment
        proxy_path = f"/{parsed.netloc}{parsed.path}"
        if parsed.query:
            proxy_path += f"?{parsed.query}"
        if parsed.fragment:
            proxy_path += f"#{parsed.fragment}"
            
        return proxy_path
        
    async def _rewrite_style_urls(self, html: str, base_url: str) -> str:
        """重写内联样式中的URL"""
        def replace_style_url(match):
            original_url = match.group(1)
            rewritten_url = self._rewrite_single_url(original_url, base_url)
            return f'url("{rewritten_url}")'
            
        return self.style_url_pattern.sub(replace_style_url, html)
        
    async def _rewrite_srcset(self, html: str, base_url: str) -> str:
        """重写srcset属性"""
        def replace_srcset(match):
            srcset = match.group(1)
            # srcset格式: "image.jpg 1x, image@2x.jpg 2x"
            parts = srcset.split(',')
            rewritten_parts = []
            
            for part in parts:
                part = part.strip()
                if ' ' in part:
                    url, descriptor = part.rsplit(' ', 1)
                else:
                    url = part
                    descriptor = ''
                    
                rewritten_url = self._rewrite_single_url(url, base_url)
                if descriptor:
                    rewritten_parts.append(f"{rewritten_url} {descriptor}")
                else:
                    rewritten_parts.append(rewritten_url)
                    
            return f'srcset="{", ".join(rewritten_parts)}"'
            
        return self.srcset_pattern.sub(replace_srcset, html)
        
    async def _handle_base_tag(self, html: str, base_url: str) -> str:
        """处理<base>标签"""
        # 如果页面有<base>标签，需要移除或修改
        # 因为它会干扰相对URL的计算
        base_pattern = re.compile(r'<base[^>]*>', re.IGNORECASE)
        return base_pattern.sub('', html)
```

    """URL规范化"""

    # /proxy/example.com/proxy/example.com/path -> /proxy/example.com/path
    duplicate_pattern = r'(/[^/]+/\1)+'
    url = re.sub(duplicate_pattern, r'\1', url)

    if url.startswith('//'):
        url = 'https:' + url
    elif not url.startswith(('http://', 'https://', '/')):
        # 相对路径
        url = urljoin(base_url, url)
        
    # 3. 移除端口默认值
    url = re.sub(r':80(/|$)', r'\1', url)
    url = re.sub(r':443(/|$)', r'\1', url)
    
    return url

import os
import mimetypes
from pathlib import Path
from typing import Optional

class PageServer:
    def __init__(self, config, logger):
        self.config = config
        self.logger = logger
        self.pages_dir = Path('pages')
        self.routes = config.get('pageRoutes', {})
        
        # 初始化MIME类型
        mimetypes.init()
        
    async def handle_request(self, path: str) -> Optional[tuple]:
        """处理静态文件请求"""
        # 1. 匹配路由
        route_name = self._match_route(path)
        if not route_name:
            return None
            
        # 2. 构建文件路径
        file_path = self._build_file_path(path, route_name)
        
        # 3. 安全检查
        if not self._is_safe_path(file_path):
            self.logger.warning(f"路径遍历攻击尝试: {path}")
            return None
            
        # 4. 读取文件
        if not file_path.exists():
            return None
            
        if file_path.is_dir():
            file_path = file_path / 'index.html'
            
        if not file_path.exists():
            return None
            
        # 5. 读取内容
        content = file_path.read_bytes()
        
        # 6. 确定MIME类型
        mime_type = self._get_mime_type(file_path)
        
        return content, mime_type
        
    def _match_route(self, path: str) -> Optional[str]:
        """匹配路由"""
        # 精确匹配
        if path in self.routes:
            return self.routes[path]
            
        # 前缀匹配
        for route_pattern, route_name in self.routes.items():
            if path.startswith(route_pattern):
                return route_name
                
        return None
        
    def _build_file_path(self, path: str, route_name: str) -> Path:
        """构建文件路径"""
        # 移除路由前缀
        relative_path = path
        for route_pattern in self.routes.keys():
            if path.startswith(route_pattern):
                relative_path = path[len(route_pattern):]
                break
                
        # 构建完整路径
        return self.pages_dir / route_name / relative_path.lstrip('/')
        
    def _is_safe_path(self, file_path: Path) -> bool:
        """检查路径安全性（防止目录遍历）"""
        try:
            # 解析真实路径
            real_path = file_path.resolve()
            pages_real_path = self.pages_dir.resolve()
            
            # 检查是否在pages目录内
            return str(real_path).startswith(str(pages_real_path))
        except (OSError, ValueError):
            return False
            
    def _get_mime_type(self, file_path: Path) -> str:
        """获取MIME类型"""
        mime_type, _ = mimetypes.guess_type(str(file_path))
        return mime_type or 'application/octet-stream'
```

async def handle_large_file(self, file_path: Path, writer):
    """流式传输大文件"""
    chunk_size = 65536  # 64KB
    file_size = file_path.stat().st_size
    
    # 发送响应头
    writer.write(b"HTTP/1.1 200 OK\r\n")
    writer.write(f"Content-Length: {file_size}\r\n".encode())
    writer.write(f"Content-Type: {self._get_mime_type(file_path)}\r\n".encode())
    writer.write(b"\r\n")
    
    # 流式读取并发送
    with open(file_path, 'rb') as f:

            chunk = f.read(chunk_size)
            if not chunk:
                break
            writer.write(chunk)
            await writer.drain()

### 2.9 优雅退出模块 (modules/exit.py)

#### 功能说明

实现优雅退出机制，确保：

- 完成正在处理的请求
- 保存配置和缓存
- 关闭所有连接

#### 技术实现

```python
import signal
import asyncio
from typing import Set
import atexit

class GracefulExit:
    shutdown_event = None
    logger = None
    active_tasks: Set[asyncio.Task] = set()
    
    @classmethod
    def setup(cls, shutdown_event: asyncio.Event, logger):
        """设置优雅退出"""
        cls.shutdown_event = shutdown_event
        cls.logger = logger
        
        # 注册信号处理器
        signal.signal(signal.SIGINT, cls._signal_handler)
        signal.signal(signal.SIGTERM, cls._signal_handler)
        
        # Windows不支持SIGQUIT
        try:
            signal.signal(signal.SIGQUIT, cls._signal_handler)
        except AttributeError:
            pass
            
        # 注册退出函数
        atexit.register(cls._cleanup)
        
    @classmethod
    def _signal_handler(cls, signum, frame):
        """信号处理器"""
        signal_names = {
            signal.SIGINT: 'SIGINT',
            signal.SIGTERM: 'SIGTERM',
        }
        
        signal_name = signal_names.get(signum, f'Signal {signum}')
        cls.logger.info(f"接收到信号 {signal_name}，准备优雅退出")
        
        if cls.shutdown_event:
            cls.shutdown_event.set()
            
    @classmethod
    def register_task(cls, task: asyncio.Task):
        """注册活动任务"""
        cls.active_tasks.add(task)
        task.add_done_callback(cls.active_tasks.discard)
        
    @classmethod
    async def wait_for_tasks(cls, timeout: int = 30):
        """等待所有任务完成"""
        if not cls.active_tasks:
            return
            
        cls.logger.info(f"等待 {len(cls.active_tasks)} 个任务完成...")
        
        try:
            await asyncio.wait_for(
                asyncio.gather(*cls.active_tasks, return_exceptions=True),
                timeout=timeout
            )
        except asyncio.TimeoutError:
            cls.logger.warning(f"等待超时，仍有 {len(cls.active_tasks)} 个任务未完成")
            
    @classmethod
    def _cleanup(cls):
        """清理资源"""
        cls.logger.info("执行清理操作...")
        # 保存配置、缓存等
```

async def process_response(content: bytes, content_type: str, 
                          target_url: str, config) -> bytes:
    """完整的响应处理流程"""
    
    # 1. 解压缩
    content = await decompress(content, encoding)
    
    # 2. 检测字符集
    charset = detect_charset(content, content_type)
    
    # 3. 解码为文本
    text = content.decode(charset)
    
    # 4. 补全相对URL
    text = complete_relative_urls(text, target_url)
    
    # 5. 修正绝对URL
    text = rewrite_absolute_urls(text, target_url, config)
    
    # 6. 清理重复代理路径
    text = normalize_proxy_urls(text)
    
    # 7. 重新编码
    content = text.encode(charset)
    
    # 8. 重新压缩
    content = await compress(content, 'gzip')
    
    return content

### 4.2 压缩与解压缩

```python
import gzip
import zlib
from typing import Tuple, Optional

async def decompress(content: bytes, encoding: Optional[str]) -> bytes:
    """解压缩内容"""
    if not encoding:
        return content
        
    if encoding == 'gzip':
        return gzip.decompress(content)
    elif encoding == 'deflate':
        return zlib.decompress(content)
    else:
        # 不支持的压缩格式，返回原始内容
        return content
        
async def compress(content: bytes, method: str = 'gzip', 
                  level: int = 6) -> Tuple[bytes, str]:
    """压缩内容"""
    if method == 'gzip':
        return gzip.compress(content, compresslevel=level), 'gzip'
    elif method == 'deflate':
        return zlib.compress(content, level), 'deflate'
    else:
        return content, 'identity'
```

### 4.3 相对URL补全算法

```python
from urllib.parse import urlparse, urljoin

class URLCompleter:
    def __init__(self):
        self.redirect_history = []
        
    def complete_relative_url(self, relative_url: str, 
                             current_url: str, 
                             referer: str = None) -> str:
        """补全相对URL"""
        # 1. 已经是绝对URL
        if relative_url.startswith(('http://', 'https://')):
            return relative_url
            
        # 2. 协议相对URL
        if relative_url.startswith('//'):
            parsed = urlparse(current_url)
            return f"{parsed.scheme}:{relative_url}"
            
        # 3. 根路径相对URL
        if relative_url.startswith('/'):
            parsed = urlparse(current_url)
            return f"{parsed.scheme}://{parsed.netloc}{relative_url}"
            
        # 4. 相对路径URL
        base_url = referer if referer else current_url
        return urljoin(base_url, relative_url)
        
    def handle_redirect(self, from_url: str, to_url: str):
        """处理重定向，更新历史"""
        self.redirect_history.append((from_url, to_url))
        
        # 限制历史长度
        if len(self.redirect_history) > 20:
            self.redirect_history.pop(0)
            
    def get_base_url_for_redirect(self) -> str:
        """根据重定向历史计算基础URL"""
        if self.redirect_history:
            return self.redirect_history[-1][1]
        return None
```

### 5.5 安全防护

```python
class SecurityMiddleware:
    def __init__(self, config, logger):
        self.config = config
        self.logger = logger
        self.request_counts = {}  # IP -> count
        self.blocked_ips = set()
        
    async def check_request(self, client_ip: str) -> bool:
        """检查请求是否允许"""
        # 1. 检查黑名单
        if client_ip in self.blocked_ips:
            return False
            
        # 2. 频率限制
        if not self._check_rate_limit(client_ip):
            self.logger.warning(f"IP {client_ip} 触发频率限制")
            return False
            
        # 3. 更新计数
        self._update_request_count(client_ip)
        
        return True
        
    def _check_rate_limit(self, client_ip: str) -> bool:
        """检查频率限制"""
        max_requests = self.config.get('security.maxRequestsPerMinute', 100)
        return self.request_counts.get(client_ip, 0) < max_requests
        
    def _update_request_count(self, client_ip: str):
        """更新请求计数"""
        self.request_counts[client_ip] = self.request_counts.get(client_ip, 0) + 1
        
    def cleanup_old_counts(self):
        """定期清理旧计数"""
        # 每分钟清理一次
        self.request_counts.clear()
```
